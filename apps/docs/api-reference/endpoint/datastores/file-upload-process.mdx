### Path

<ParamField path="datastoreId" type="string" required>
  The id of the datastore where you want to upload a file.
</ParamField>

### Body

<ParamField body="id" type="string" required>
  ID of the upload link generated with /datastores/file-upload/generate-link/`<datastoreId>`
</ParamField>

### Response

<ResponseField name="id" type="string">
  Id of the upload link.
</ResponseField>

--header 'Content-Type: application/json' \
